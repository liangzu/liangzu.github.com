1 Using enumerate()

cities = ['Marseille', 'Amsterdam', 'New York', 'London']
# the bad way
i = 0
for city in cities:
    print(i, city)
    i += 1
# the good way
for i, city in enumerate(cities):
    print(i, city)

2 Using zip()

x_list = [1,2,3]
y_list = [2,4,6]
# the bad way
for i in range(len(x_list)):
    x = x_list[i]
    y = y_list[i]
    print(x, y)
# the good way
for x, y in zip(x_list, y_list):
    print(x, y)

3 Default dict values

ages = {
    'Mary': 31,
    'Jonathan': 28
}
# the bad way
if 'Dick' in ages:
    age = ages['Dick']
else:
    age = 'Unknown'

# the good way
age = ages.get('Dick', 'Unknown')

4 For … else

needle = 'd'
haystack = ['a', 'b', 'c', 'd']
# the bad way
found = False
for letter in haystack:
    if needle == letter:
        print('Found!')
        found = True
        break

if not found:
    print('Not found!')

# the good way
for letter in haystack:
    if needle == letter:
        print('Found!')

        break
else:
    print('Not found!')

5 File reading with 'with'

# the bad way
f = open('pimpin-aint-easy.txt')
text = f.read()
for line in text.split('\n'):
    print(line)
f.close()
# the better way
f = open('pimpin-aint-easy.txt')
for line in f:
    print(line)
f.close()
# a even better way
with open('pimpin-aint-easy.txt') as f:
    for line in f:
        print(line)

6 Exception handling

print('Converting!')
try:
    print(int('x'))
except:
    print('Conversion failed!')
else:  # if no-except
    print('Conversion successful')
finally:  # always
    print('Done!')
